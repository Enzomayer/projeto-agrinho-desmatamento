public class RelacaoCampoCidade { public static void main(String[] args) { Campo campo = new Campo("Produção agrícola", "Fornece alimentos e matérias-primas para a cidade."); Cidade cidade = new Cidade("Indústria e comércio", "Processa produtos do campo e oferece serviços."); System.out.println("Relação entre Campo e Cidade:\n"); campo.exibirInformacoes(); cidade.exibirInformacoes(); System.out.println("\nInterações:"); System.out.println("- O campo fornece recursos naturais para a cidade."); System.out.println("- A cidade fornece produtos industrializados e serviços para o campo."); System.out.println("- Migrações ocorrem entre campo e cidade em busca de melhores condições de vida."); } } class Campo { private String caracteristica; private String contribuicao; public Campo(String caracteristica, String contribuicao) { this.caracteristica = caracteristica; this.contribuicao = contribuicao; } public void exibirInformacoes() { System.out.println("Campo:"); System.out.println("- Característica: " + caracteristica); System.out.println("- Contribuição: " + contribuicao); } } class Cidade { private String caracteristica; private String contribuicao; public Cidade(String caracteristica, String contribuicao) { this.caracteristica = caracteristica; this.contribuicao = contribuicao; } public void exibirInformacoes() { System.out.println("Cidade:"); System.out.println("- Característica: " + caracteristica); System.out.println("- Contribuição: " + contribuicao); } }
